//----- Variables -----------------------------------------------------------------------------------------------------

cloneid = 10



//----- Green flag events ---------------------------------------------------------------------------------------------

WhenGreenFlagClicked()
{
    Motion.GoToXY(-190, -165);
    Control.Wait(0.5);
    Call Clones;
    Forever
    {
        If (HEALTH < 0)
        {
            Costume.GoToFrontBack(front);
        }
    }
}


//----- Start as clone ------------------------------------------------------------------------------------------------

Control.WhenIStartAsClone()
{
    Costume.MoveLayers(backward, 2);
    Forever
    {
        Looks.SwitchCostumeTo(Operator.Join("costume", Operator.LetterOf(Score, cloneid)));
        Costume.MoveLayers(forward, 1);
        If (HEALTH < 0)
        {
            Costume.GoToFrontBack(front);
        }
    }
}


//----- Custom blocks -------------------------------------------------------------------------------------------------

Define Clones (warp=true)
{
    Looks.Show();
    cloneid = 0;
    Repeat (10)
    {
        cloneid += 1;
        Motion.ChangeXBy(15);
        Control.CreateCloneOf(_myself_);
        Control.Wait(0);
    }
    Motion.SetX(-205);
    Looks.SwitchCostumeTo(costume1s);
}
